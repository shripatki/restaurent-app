  getMenuById(id:string){
    return {...this.menu.find(p => p.restaurantId===id)};
  }
  getItemDetails(name:string){
    return {...this.menu.find(p=>{
      p.menuItems.find(item =>{
        item.name==name;
      });
    })};
  }
